```c++
## 🔍 리팩토링 전 (문제점)
- 기존에는 vendingMachine과 BreackableObject에 아이템 드랍 로직을 두었다. 하지만 기존 Vending
  Machine에 스폰은 타입만 랜덤으로 정해 뽑는 방식이였는데 각 ItemType중 드랍할 타입을 뽑고 이 중
  각 Rarity(Common, Rare, Epic)마다 가중치 중심에 랜덤 드랍을 하는 식으로 개선하려고 하였다.
  ItemDrop을 하는 오브젝트들에 부모 클래스를 상위에 만들까도 고민했지만 이미 InteractableSturucture
  라는 추상클래스를 상속받고 있어 다중 상속이 안될뿐만 아니라 상속 관계가 너무 복잡해 질거 같았다.

## ✅ 리팩토링 후 (개선 내용)
- ItemDropper를 제작해 아이템 드랍을 하는 모든 오브젝트에 컴포넌트로 붙히고 이걸 참조해서 아이템
  드랍을 하도록 구현하였다. ItemDropper는 ItemTypeDropConfig를 기반으로 어떤 아이템종류를 뽑을
  지를 정한 후 Rarity를 가중치 기반으로 뽑고 정해진 타입과 희귀도에 아이템중 랜덤한 아이템을 드랍
  하도록 하였다. 드랍 애니메이션또한 from 위치에서 to 위치로 선형적이지 않고 러프하게 이동하도록
  개선 하였다.

## 💡 배운 점 / 메모
- 멘토링 전에 여러 오브젝트들을 찍다시피 만들다 보니  생각보다 중복되는 코드나 확장성이 안 좋은 
  구조들을 쉽게 찾을 수 있었다. 
- 나중에 상호작용이 가능하면서 체력을 가지고 있는 오브젝트는 어떻게 만들지를 고민하고 있었는데 
  클로드가 YAGNI (You Aren't Gonna Need It) 원칙을 알려줘서 알게 되어 리펙토링을 끝낼 수
  있었다.
```

```c#
public class ItemDropper : MonoBehaviour
{
    [Inject]
    private PoolManagerMono _poolManagerMono;

    [SerializeField] private ItemDataBaseSO itemDB;
    [SerializeField] private PoolItemSO previewItemPool;
    [SerializeField] private List<ItemTypeDropConfig> dropConfigs;

    [Header("Drop Settings")]
    [SerializeField] private float arcHeight = 1f;
    [SerializeField] private float dropDuration = 0.4f;

    public void Drop(Vector3 from, Vector3 to)
    {
        if (dropConfigs == null || dropConfigs.Count == 0) return;

        var config = dropConfigs[Random.Range(0, dropConfigs.Count)];
        if (config.rarityWeights == null || config.rarityWeights.Count == 0) return;

        var rarity = GetWeightedRarity(config.rarityWeights);
        var items = itemDB.GetRandomItems(config.type, rarity, 1);
        if (items == null || items.Count == 0) return;

        var createData = items[0].CreateItem();
        var item = _poolManagerMono.Pop<PreviewItem>(previewItemPool);
        if (item == null) return;

        item.Discard(from, createData.Item, createData.Stack);

        Sequence seq = DOTween.Sequence();
        seq.Append(item.transform.DOMoveX(to.x, dropDuration).SetEase(Ease.Linear));
        seq.Join(item.transform.DOMoveZ(to.z, dropDuration).SetEase(Ease.Linear));
        seq.Join(item.transform.DOMoveY(from.y + arcHeight, dropDuration * 0.4f).SetEase(Ease.OutQuad));
        seq.Insert(dropDuration * 0.4f, item.transform.DOMoveY(to.y, dropDuration * 0.6f).SetEase(Ease.InQuad));
        seq.OnComplete(() => item.transform.DOPunchScale(Vector3.one * 0.15f, 0.15f, 1, 0f));
    }

    private Rarity GetWeightedRarity(List<RarityWeight> rarityWeights)
    {
        float total = 0f;
        foreach (var rw in rarityWeights)
            total += rw.weight;

        float roll = Random.Range(0f, total);
        float current = 0f;

        foreach (var rw in rarityWeights)
        {
            current += rw.weight;
            if (roll <= current) return rw.rarity;
        }

        return rarityWeights[rarityWeights.Count - 1].rarity;
    }
}
```

```c#
public class VendingMachine : InteractableStructure
{
    [Header("Reference")]
    [SerializeField] private MeshRenderer meshRenderer;
    [SerializeField] private Transform discardPoint;
    [SerializeField] private Transform spawnPoint;
    [SerializeField] private ItemDropper itemDropper;
    
    [Header("Settings")]
    [SerializeField] private int minDropCount = 1;
    [SerializeField] private int maxDropCount = 5;
    [SerializeField] private float offDuration = 1.6f;
    [SerializeField] private float discardRange = 0.5f;
    [SerializeField] private float interactCooldown = 1f;

    private int _dropCount = 0;
    private bool _isOff = false;
    private float _lastInteractTime = float.NegativeInfinity;
    private Material _material;

    protected override void Awake()
    {
        base.Awake();
        _material = meshRenderer.material;
    }

    protected override void Start()
    {
        base.Start();
        Init();
    }

    private void Init()
    {
        _material.EnableKeyword("_EMISSION");
        _material.SetColor("_EmissionColor", Color.white * 10.0f);
        _isOff = false;
        _dropCount = 0;
    }

    public override void Interact(Entity interactor)
    {
        if (_isOff) return;
        if (Time.time - _lastInteractTime < interactCooldown) return;
        if(_dropCount > maxDropCount) return;
        if (_dropCount > 1)
        {
            float randomValue = Random.value; 
            if (randomValue < 0.5f)           
            {
                _dropCount = maxDropCount;
                StartCoroutine(MachineOffCoroutine(offDuration));
                return;
            }
        }

        _dropCount++;
        _lastInteractTime = Time.time;

        Vector3 to = discardPoint.position;
        to.x += Random.Range(-discardRange, discardRange);
        to.z += Random.Range(-discardRange, discardRange);
        itemDropper.Drop(spawnPoint.position, to);
    }

    private IEnumerator MachineOffCoroutine(float duration)
    {
        Color startColor = _material.GetColor("_EmissionColor");
        Color endColor = Color.white * 0f;
        float elapsed = 0f;

        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            _material.SetColor("_EmissionColor", Color.Lerp(startColor, endColor, elapsed / duration));
            yield return null;
        }

        _material.SetColor("_EmissionColor", endColor);
        _material.DisableKeyword("_EMISSION");
    }
}
```
